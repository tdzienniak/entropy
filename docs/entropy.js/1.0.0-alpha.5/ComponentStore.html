register(descriptor)

Source:
  • ComponentStore.js, line 63

Registers component.

Example
game.component.register({
  type: 'Position',
  onCreate(x, y) {
    this.x = x;
    this.y = y;
  },
});

const c = game.component.create('Position', 1, 2);
c.x; // 1
c.y; // 2
